Modules

1. AMD loader

  • require

2. Creating module

// app/counter.js
define(function () {
    var privateValue = 0;

    return {
        increment: function () {
            privateValue++;
        },

        decrement: function () {
            privateValue--;
        },

        getValue: function () {
            return privateValue;
        }
    };
});

3. Loading module

require([
    "app/counter"
], function (counter) {
    log(counter.getValue()); // 0
    counter.increment();
    log(counter.getValue()); // 1
    counter.decrement();
    log(counter.getValue()); // 0
});

4. Modules loading Modules

define([
    "dojo/_base/declare",
    "dojo/dom",
    "app/dateFormatter"
], function (declare, dom, dateFormatter) {
    return declare(null, {
        showDate: function (id, date) {
            dom.byId(id).innerHTML = dateFormatter.format(date);
        }
    });
});
